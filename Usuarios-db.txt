Crear tablas de usuarios y productos:

CREATE TABLE users (
  id SERIAL PRIMARY KEY,
  name TEXT,
  email TEXT UNIQUE,
  password TEXT
);

CREATE TABLE productos (
  id SERIAL PRIMARY KEY,
  usuario_id INTEGER REFERENCES users(id),
  nombre TEXT,
  precio NUMERIC,
  stock INTEGER,
  caracteristicas JSONB
);
